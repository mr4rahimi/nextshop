آماده‌سازی پروژه لوکال

فایل‌های env و تنظیمات لازم قبل از build

💡قبل از هر کار، فایل‌های env رو آماده کن. هرگز .env.local رو به سرور ارسال نکن.

فایل .env.production رو در root پروژه بساز:

.env.production
# آدرس سایت — بدون / آخر
NEXT_PUBLIC_SITE_URL=https://yourdomain.com
NEXT_PUBLIC_STORE_NAME=نام فروشگاه

# دیتابیس — بعداً با اطلاعات سرور پر می‌کنی
DATABASE_URL=postgresql://USER:PASS@localhost:5432/DB_NAME

# JWT — یه رشته تصادفی قوی بذار
JWT_SECRET=your_super_secret_jwt_key_min_32_chars

NODE_ENV=production

# سرویس‌های جانبی (اختیاری)
# IRANPAYAMAK_API_KEY=...

فایل next.config.ts رو تمیز کن:

next.config.ts
import type { NextConfig } from "next";

const nextConfig: NextConfig = {};

export default nextConfig;

فایل app/auth/page.tsx — مطمئن شو return درست باشه:

app/auth/page.tsx
import { Suspense } from "react";
import AuthClient from "@/components/auth/AuthClient";

export const metadata = { title: "ورود | فروشگاه" };

export default function AuthPage() {
  return (
    <Suspense fallback={null}>
      <AuthClient />
    </Suspense>
  );
}

Build و بسته‌بندی

ساخت فایل production و آرشیو برای آپلود

bash — لوکال
# تست build لوکال
pnpm build

# اگه build موفق بود، آرشیو کامل بساز
cd /path/to/your/project

tar --exclude='node_modules/.cache' \
    --exclude='.git' \
    --exclude='.next/cache' \
    --exclude='.env.local' \
    -czf /tmp/shop-full.tar.gz \
    node_modules .next public prisma lib app \
    components types package.json pnpm-lock.yaml \
    next.config.ts tsconfig.json postcss.config.mjs \
    tailwind.config.ts .env.production prisma.config.ts \
    prisma.config.mjs

# چک حجم
du -sh /tmp/shop-full.tar.gz
⚠️اگه node_modules بیش از ۳۰۰MB هست، می‌تونی بدون node_modules آرشیو کنی و روی سرور pnpm install بزنی — اما به اینترنت آزاد نیاز داری.

Node.js 20 روی سرور ایران

نصب Node از طریق binary — بدون نیاز به اینترنت آزاد

🇮🇷سرورهای ایران به npmjs.org و nodejs.org دسترسی ندارن. باید binary رو دانلود کنی و آپلود کنی.

لوکال — دانلود binary:

bash — لوکال
wget https://nodejs.org/dist/v20.19.0/node-v20.19.0-linux-x64.tar.xz

فایل رو به یه هاست آپلود کن، بعد روی سرور:

bash — سرور
# دانلود از هاست شخصی
wget https://yourhost.com/node-v20.19.0-linux-x64.tar.xz -O /tmp/node20.tar.xz

# نصب
cd /usr/local
tar -xJf /tmp/node20.tar.xz --strip-components=1

# تأیید
node -v  # باید v20.x.x باشه
npm -v

نصب pnpm و PM2:

bash — سرور
# تنظیم میرور لیارا برای npm
npm config set registry https://package-mirror.liara.ir/repository/npm/ --global

# نصب pnpm نسخه 9
npm install -g pnpm@9

# نصب PM2
npm install -g pm2

# تأیید
pnpm -v && pm2 -v

PostgreSQL و دیتابیس

ساخت user و database اختصاصی

bash — سرور
# ساخت user و database
sudo -u postgres psql << 'EOF'
CREATE USER myuser WITH PASSWORD 'StrongPassword123!';
CREATE DATABASE mydb OWNER myuser;
GRANT ALL PRIVILEGES ON DATABASE mydb TO myuser;
EOF

# تست اتصال
sudo -u postgres psql -c "\l" | grep mydb
⚠️رمز عبور قوی انتخاب کن. DATABASE_URL رو در .env سرور آپدیت کن: postgresql://myuser:StrongPassword123!@localhost:5432/mydb

آپلود فایل‌ها به سرور

انتقال پروژه از لوکال به سرور

روش ۱ — مستقیم با scp:

bash — لوکال
scp /tmp/shop-full.tar.gz root@SERVER_IP:/tmp/

روش ۲ — از طریق هاست واسط (برای سرورهای ایران):

bash — سرور
# دانلود از هاست شخصی
wget https://yourhost.com/shop-full.tar.gz -O /tmp/shop-full.tar.gz

# extract
mkdir -p /var/www/mymonta
tar -xzf /tmp/shop-full.tar.gz -C /var/www/mymonta/

# تأیید
ls /var/www/mymonta/

آپدیت .env روی سرور:

bash — سرور
# ویرایش DATABASE_URL در فایل .env
sed -i 's|DATABASE_URL=.*|DATABASE_URL="postgresql://myuser:StrongPassword123!@localhost:5432/mydb"|' \
  /var/www/mymonta/.env

# حذف .env.local که ممکنه override کنه
rm -f /var/www/mymonta/.env.local

# تأیید
grep DATABASE_URL /var/www/mymonta/.env

نصب dotenv اگه موجود نبود:

bash — سرور
npm install dotenv --registry="https://package-mirror.liara.ir/repository/npm/"
cp -r /root/node_modules/dotenv /var/www/mymonta/node_modules/

Build روی سرور و راه‌اندازی PM2

build مجدد با env سرور و شروع سرویس

💡برای اطمینان، build رو روی خود سرور هم بگیر تا env variables درست بیلد بشن.
bash — سرور
cd /var/www/mymonta

# Build روی سرور
NODE_ENV=production \
DATABASE_URL="postgresql://myuser:StrongPassword123!@localhost:5432/mydb" \
./node_modules/.bin/next build

# شروع با PM2
pm2 start node_modules/.bin/next \
  --name "myshop" \
  --interpreter none \
  -- start -p 3000

# ذخیره و auto-start
pm2 save
pm2 startup

# وضعیت
pm2 status

تنظیم Nginx

Reverse proxy و serve فایل‌های static

bash — سرور
# حذف default
rm -f /etc/nginx/sites-enabled/default

# ساخت config جدید
cat > /etc/nginx/sites-available/myshop << 'NGINX'
server {
    listen 80;
    server_name yourdomain.com www.yourdomain.com;

    client_max_body_size 50M;

    location /_next/static/ {
        alias /var/www/mymonta/.next/static/;
        expires 1y;
        add_header Cache-Control "public, immutable";
    }

    location /assets/ {
        alias /var/www/mymonta/public/assets/;
        expires 30d;
    }

    location /uploads/ {
        alias /var/www/mymonta/public/uploads/;
        expires 30d;
    }

    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }
}
NGINX

# فعال‌سازی
ln -s /etc/nginx/sites-available/myshop /etc/nginx/sites-enabled/
nginx -t && systemctl restart nginx

SSL و HTTPS

نصب گواهینامه — دو روش

روش ۱ — Let's Encrypt (اگه سرور به اینترنت آزاد دسترسی داره):

bash — سرور
apt install -y certbot python3-certbot-nginx
certbot --nginx -d yourdomain.com -d www.yourdomain.com

روش ۲ — Certificate دستی (از CDN یا cPanel):

bash — سرور
# ساخت پوشه
mkdir -p /etc/nginx/ssl/mysite

# ذخیره فایل‌ها (محتوا رو paste کن)
nano /etc/nginx/ssl/mysite/fullchain.pem
nano /etc/nginx/ssl/mysite/privkey.pem

آپدیت Nginx config برای HTTPS:

nginx config با SSL
server {
    listen 80;
    server_name yourdomain.com www.yourdomain.com;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl;
    server_name yourdomain.com www.yourdomain.com;

    ssl_certificate     /etc/nginx/ssl/mysite/fullchain.pem;
    ssl_certificate_key /etc/nginx/ssl/mysite/privkey.pem;
    ssl_protocols       TLSv1.2 TLSv1.3;
    ssl_ciphers         HIGH:!aNULL:!MD5;

    client_max_body_size 50M;

    add_header Content-Security-Policy "default-src * 'unsafe-inline' 'unsafe-eval' data: blob:;" always;

    location /_next/static/ {
        alias /var/www/mymonta/.next/static/;
        expires 1y;
        add_header Cache-Control "public, immutable";
    }

    location /assets/ {
        alias /var/www/mymonta/public/assets/;
        expires 30d;
    }

    location /uploads/ {
        alias /var/www/mymonta/public/uploads/;
        expires 30d;
    }

    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }
}

بروزرسانی سایت

فرآیند دیپلوی نسخه‌های جدید

bash — لوکال (هر بار)
# build جدید
pnpm build

# آرشیو — شامل node_modules هم بشه (اول بار یا وقتی پکیج جدید داری)
tar --exclude='node_modules/.cache' \
    --exclude='.git' \
    --exclude='.next/cache' \
    -czf /tmp/shop-update.tar.gz \
    node_modules .next app components lib prisma \
    public types package.json pnpm-lock.yaml \
    next.config.ts tsconfig.json .env.production \
    prisma.config.ts prisma.config.mjs

# یا فقط کدها (بدون node_modules — سریع‌تر، اگه پکیج جدید نداری)
tar -czf /tmp/shop-update.tar.gz \
    .next app components lib prisma public types \
    next.config.ts tsconfig.json package.json pnpm-lock.yaml

# آپلود به هاست و بعد روی سرور:
bash — سرور
wget https://yourhost.com/shop-update.tar.gz -O /tmp/shop-update.tar.gz
tar -xzf /tmp/shop-update.tar.gz -C /var/www/mymonta/

# اگه migration جدید داری
cd /var/www/mymonta
DATABASE_URL="postgresql://myuser:pass@localhost:5432/mydb" \
./node_modules/.bin/prisma migrate deploy

# rebuild روی سرور
NODE_ENV=production \
DATABASE_URL="postgresql://myuser:pass@localhost:5432/mydb" \
./node_modules/.bin/next build

# اگه app/page.tsx وجود نداره یا تغییر کرده — چک کن
ls /var/www/mymonta/app/page.tsx || \
  cp "/var/www/mymonta/app/(shop)/page.tsx" /var/www/mymonta/app/page.tsx

# پوشه‌های قدیمی که نباید باشن رو حذف کن
rm -rf /var/www/mymonta/app/store 2>/dev/null || true
rm -rf /var/www/mymonta/app/checkout 2>/dev/null || true

# restart
pm2 restart mymonta-shop --update-env
pm2 save
✅سایت بروز شد! با pm2 logs myshop --lines 20 وضعیت رو چک کن.
⚠️بعد از آپدیت چک کن: app/page.tsx باید Header/Footer داشته باشه. اگه صفحه اصلی بدون هدر بود، فایل app/page.tsx رو با نسخه کامل جایگزین کن.
